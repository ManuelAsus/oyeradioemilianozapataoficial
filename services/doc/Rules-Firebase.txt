rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Lectura pública para todo
    match /{document=**} {
      allow read: if true;
    }

    // Colecciones administrativas — escritura solo para admin autenticado
    match /schedule/{doc}    { allow write: if request.auth != null; }
    match /news/{doc}        { allow write: if request.auth != null; }
    match /announcers/{doc}  { allow write: if request.auth != null; }
    match /sponsors/{doc}    { allow write: if request.auth != null; }
    match /directory/{doc}   { allow write: if request.auth != null; }
    match /streaming/{doc}   { allow write: if request.auth != null; }
    match /settings/{doc}    { allow write: if request.auth != null; }
    match /social/{doc}      { allow write: if request.auth != null; }
    match /users/{doc}       { allow write: if request.auth != null; }

    // Mensajes de contacto — escritura pública
    match /contactMessages/{doc} {
      allow read:  if request.auth != null;
      allow write: if true;
    }

    // Señalización WebRTC (liveStream) — escritura pública necesaria
    // Los visitantes deben poder registrarse como espectadores sin autenticarse
    match /liveStream/{document=**} {
      allow read, write: if true;
    }
  }
}
